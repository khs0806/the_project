year=[2015,2016,2017,2018]
plt.rcParams["figure.figsize"]=(20,20)
name=['주거','상업','공업','녹지']
for j in year:
    for i in range(0,len(grouped.get_group(j))):
        plt.subplot(7,4,i+1)
        plt.pie([grouped.get_group(j)['주거비율'].iloc(0)[i],
                 grouped.get_group(j)['상업비율'].iloc(0)[i],
                grouped.get_group(j)['공업비율'].iloc(0)[i],
                grouped.get_group(j)['녹지비율'].iloc(0)[i]],
#                 labels=name
               )
        plt.legend(name)
        plt.title(str(j)+' '+grouped.get_group(j)['자치구'].iloc(0)[i])
        plt.savefig('{}pie.png'.format(j))

    plt.show()
