In [4]:
y = data.geo_y.values
x = data.geo_x.values
locations = list(zip(y,x))
popups = list(data.title.values)

icon_create_function = """\
function(cluster) {
    return L.divIcon({
    html: '<b>' + cluster.getChildCount() + '</b>',
    className: 'marker-cluster marker-cluster-large',
    iconSize: new L.Point(30, 30)
    });
}"""

map_mw2 = folium.Map(location=[37.557376, 126.904376], zoom_start = 16)

marker_cluster = MarkerCluster(
    locations=locations, popups=popups,
    name='seoul',
    overlay=True,
    control=True,
    icon_create_function=icon_create_function
)

marker_cluster.add_to(map_mw2)
folium.LayerControl().add_to(map_mw2)
